int numIslands(char** grid, int gridSize, int* gridColSize) {
    int m = gridSize, n = *gridColSize;

    // m x n grid
    int i, j;
    int islands = 0;
    for (i = 0; i < m; i++)
        for (j = 0; j < n; j++) {
            if (grid[i][j] == '1') {
                dfs(grid, m, n, i, j); 
                islands++;
            }   
        }   
    return islands;
}

void dfs(char** grid, int m, int n, int i, int j) {
    if (i < 0 || i >= m || j < 0 || j >= n) {
        return;
    }   
    if (grid[i][j] == '0') return;

    grid[i][j] = '0';
    dfs(grid, m, n, i - 1, j); 
    dfs(grid, m, n, i + 1, j); 
    dfs(grid, m, n, i, j - 1); 
    dfs(grid, m, n, i, j + 1); 
}
